   ```typescript
   import { PrismaClient } from '@prisma/client'
   import { fieldEncryptionExtension } from 'prisma-field-encryption'

   const globalForPrisma = globalThis as unknown as {
     prisma: ReturnType<typeof createPrismaClient> | undefined
   }

   function createPrismaClient() {
     const base = new PrismaClient({
       log: process.env.NODE_ENV === 'development' ? ['query', 'error', 'warn'] : ['error'],
     })
     return base.$extends(
       fieldEncryptionExtension({
         encryptionKey: process.env.ENCRYPTION_KEY!,
       })
     )
   }

   export const prisma = globalForPrisma.prisma ?? createPrismaClient()

   if (process.env.NODE_ENV !== 'production') {
     globalForPrisma.prisma = prisma
   }
   ```

4. Create `src/lib/r2.ts` — R2 client and presigned URL helpers:
   ```typescript
   import { S3Client, PutObjectCommand, DeleteObjectCommand } from '@aws-sdk/client-s3'
   import { getSignedUrl } from '@aws-sdk/s3-request-presigner'
   import { randomUUID } from 'crypto'

   export const r2 = new S3Client({
     region: 'auto',
     endpoint: `https://${process.env.R2_ACCOUNT_ID}.r2.cloudflarestorage.com`,
     credentials: {
       accessKeyId: process.env.R2_ACCESS_KEY_ID!,
       secretAccessKey: process.env.R2_SECRET_ACCESS_KEY!,
     },
   })

   /**
    * Generate a presigned PUT URL for direct client-to-R2 upload.
    * URL expires in 5 minutes — sufficient for admin uploads.
    * R2 credentials never reach the browser.
    */
   export async function generatePresignedUploadUrl(
     folder: 'sites' | 'addons',
     entityId: string,
     contentType: string
   ): Promise<{ uploadUrl: string; key: string }> {
     const ext = contentType === 'image/webp' ? 'webp' : 'jpg'
     const key = `${folder}/${entityId}/${randomUUID()}.${ext}`
     const command = new PutObjectCommand({
       Bucket: process.env.R2_BUCKET_NAME!,
       Key: key,
       ContentType: contentType,
     })
     const uploadUrl = await getSignedUrl(r2, command, { expiresIn: 300 })
     return { uploadUrl, key }
   }

   /**
    * Get the public URL for a stored R2 object key.
    */
   export function getPublicUrl(key: string): string {
     return `${process.env.R2_PUBLIC_URL}/${key}`
   }

   /**
    * Delete an object from R2 by key.
    */
   export async function deleteObject(key: string): Promise<void> {
     await r2.send(new DeleteObjectCommand({
       Bucket: process.env.R2_BUCKET_NAME!,
       Key: key,
     }))
   }
   ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| .env → runtime | Secrets (DATABASE_URL, R2 keys, JWT_SECRET, ENCRYPTION_KEY) must never be committed to git or exposed in client bundles |
| client → R2 presigned URL | Client uploads directly to R2 using a short-lived signed URL; R2 credentials never leave the server |
| Neon pooled → direct | DIRECT_URL must only be used for migrations; runtime uses pooled connection |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Information Disclosure | .env file | mitigate | Add `.env` to `.gitignore`; only commit `.env.example` |
| T-01-02 | Information Disclosure | ENCRYPTION_KEY | mitigate | Minimum 32 bytes random; store in Vercel env vars, not local git |
| T-01-03 | Tampering | R2 bucket | mitigate | Presigned URLs expire in 300s; PUT only — no DELETE via presigned URL |
| T-01-04 | Information Disclosure | customerPhone | mitigate | `/// @encrypted` annotation ensures AES encryption at rest via prisma-field-encryption (PIPA requirement) |
| T-01-05 | Elevation of Privilege | Prisma migrations | accept | Only run via DIRECT_URL on trusted CI or local dev; pooled connection cannot run migrations |
</threat_model>
