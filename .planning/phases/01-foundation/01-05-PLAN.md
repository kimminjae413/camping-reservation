   ```typescript
   import { NextRequest, NextResponse } from 'next/server'
   import { requireAdmin } from '@/lib/middleware-auth'
   import { prisma } from '@/lib/prisma'
   import { addonSchema } from '@/lib/validations/addon'

   export async function GET() {
     const authError = await requireAdmin()
     if (authError) return authError

     const addons = await prisma.addon.findMany({
       orderBy: { createdAt: 'desc' },
     })
     return NextResponse.json(addons)
   }

   export async function POST(req: NextRequest) {
     const authError = await requireAdmin()
     if (authError) return authError

     const body = await req.json()
     const parsed = addonSchema.safeParse(body)
     if (!parsed.success) {
       return NextResponse.json(
         { error: '입력값이 올바르지 않습니다', details: parsed.error.flatten() },
         { status: 400 }
       )
     }

     const addon = await prisma.addon.create({ data: parsed.data })
     return NextResponse.json(addon, { status: 201 })
   }
   ```

   ```typescript
   import { NextRequest, NextResponse } from 'next/server'
   import { requireAdmin } from '@/lib/middleware-auth'
   import { prisma } from '@/lib/prisma'
   import { deleteObject } from '@/lib/r2'
   import { addonSchema } from '@/lib/validations/addon'

   export async function GET(
     _req: NextRequest,
     { params }: { params: Promise<{ id: string }> }
   ) {
     const authError = await requireAdmin()
     if (authError) return authError

     const { id } = await params
     const addon = await prisma.addon.findUnique({ where: { id } })
     if (!addon) return NextResponse.json({ error: 'Not found' }, { status: 404 })
     return NextResponse.json(addon)
   }

   export async function PUT(
     req: NextRequest,
     { params }: { params: Promise<{ id: string }> }
   ) {
     const authError = await requireAdmin()
     if (authError) return authError

     const { id } = await params
     const body = await req.json()
     const parsed = addonSchema.safeParse(body)
     if (!parsed.success) {
       return NextResponse.json(
         { error: '입력값이 올바르지 않습니다', details: parsed.error.flatten() },
         { status: 400 }
       )
     }

     const addon = await prisma.addon.update({
       where: { id },
       data: parsed.data,
     })
     return NextResponse.json(addon)
   }

   /**
    * PATCH: toggle isActive only.
    * Used by the list page toggle switch (D-13).
    * Body: { isActive: boolean }
    */
   export async function PATCH(
     req: NextRequest,
     { params }: { params: Promise<{ id: string }> }
   ) {
     const authError = await requireAdmin()
     if (authError) return authError

     const { id } = await params
     const body = await req.json()

     if (typeof body.isActive !== 'boolean') {
       return NextResponse.json({ error: 'isActive must be boolean' }, { status: 400 })
     }

     const addon = await prisma.addon.update({
       where: { id },
       data: { isActive: body.isActive },
     })
     return NextResponse.json(addon)
   }

   export async function DELETE(
     _req: NextRequest,
     { params }: { params: Promise<{ id: string }> }
   ) {
     const authError = await requireAdmin()
     if (authError) return authError

     const { id } = await params
     const addon = await prisma.addon.findUnique({ where: { id } })
     if (!addon) return NextResponse.json({ error: 'Not found' }, { status: 404 })

     // Delete R2 photo if present
     if (addon.photoKey) {
       await deleteObject(addon.photoKey).catch(() => {
         // Log but don't fail if R2 delete fails
         console.error(`Failed to delete R2 object: ${addon.photoKey}`)
       })
     }

     await prisma.addon.delete({ where: { id } })
     return NextResponse.json({ ok: true })
   }
   ```

2. Create `src/app/(admin-shell)/admin/addons/page.tsx` — list with inline toggle (D-13):
   ```typescript
   'use client'

   import { useEffect, useState } from 'react'
   import Link from 'next/link'
   import { Button } from '@/components/ui/button'
   import { Badge } from '@/components/ui/badge'
   import { Switch } from '@/components/ui/switch'
   import {
     Table, TableBody, TableCell, TableHead, TableHeader, TableRow,
   } from '@/components/ui/table'
   import { Plus, Pencil, Trash2 } from 'lucide-react'

   const PRICE_TYPE_LABEL: Record<string, string> = {
     PER_BOOKING: '건당',
     PER_PERSON: '인당',
     PER_NIGHT: '박당',
   }

   interface Addon {
     id: string
     name: string
     category: string | null
     price: number
     priceType: string
     maxDailyQty: number
     isActive: boolean
     photoKey: string | null
   }

   export default function AddonsPage() {
     const [addons, setAddons] = useState<Addon[]>([])
     const [loading, setLoading] = useState(true)

     async function fetchAddons() {
       const res = await fetch('/api/admin/addons')
       const data = await res.json()
       setAddons(data)
       setLoading(false)
     }

     useEffect(() => { fetchAddons() }, [])

     async function handleToggle(id: string, isActive: boolean) {
       // Optimistic update
       setAddons((prev) =>
         prev.map((a) => (a.id === id ? { ...a, isActive } : a))
       )
       await fetch(`/api/admin/addons/${id}`, {
         method: 'PATCH',
         headers: { 'Content-Type': 'application/json' },
         body: JSON.stringify({ isActive }),
       })
     }

     async function handleDelete(id: string) {
       if (!confirm('이 부대시설을 삭제하시겠습니까?')) return
       await fetch(`/api/admin/addons/${id}`, { method: 'DELETE' })
       setAddons((prev) => prev.filter((a) => a.id !== id))
     }

     return (
       <div className="space-y-4">
         <div className="flex items-center justify-between">
           <div>
             <h1 className="text-2xl font-bold">부대시설</h1>
             <p className="text-sm text-muted-foreground mt-1">
               추가 서비스 항목을 관리합니다
             </p>
           </div>
           <Button asChild>
             <Link href="/admin/addons/new">
               <Plus className="mr-2 h-4 w-4" />
               부대시설 추가
             </Link>
           </Button>
         </div>

         <div className="rounded-md border">
           <Table>
             <TableHeader>
               <TableRow>
                 <TableHead>이름</TableHead>
                 <TableHead>카테고리</TableHead>
                 <TableHead>가격</TableHead>
                 <TableHead>가격 유형</TableHead>
                 <TableHead>일별 최대</TableHead>
                 <TableHead>활성 (D-13)</TableHead>
                 <TableHead className="text-right">관리</TableHead>
               </TableRow>
             </TableHeader>
             <TableBody>
               {loading ? (
                 <TableRow>
                   <TableCell colSpan={7} className="py-8 text-center text-muted-foreground">
                     로딩 중...
                   </TableCell>
                 </TableRow>
               ) : addons.length === 0 ? (
                 <TableRow>
                   <TableCell colSpan={7} className="py-8 text-center text-muted-foreground">
                     등록된 부대시설이 없습니다
                   </TableCell>
                 </TableRow>
               ) : (
                 addons.map((addon) => (
                   <TableRow key={addon.id}>
                     <TableCell className="font-medium">{addon.name}</TableCell>
                     <TableCell>
                       {addon.category ? (
                         <Badge variant="outline">{addon.category}</Badge>
                       ) : (
                         <span className="text-muted-foreground text-sm">-</span>
                       )}
                     </TableCell>
                     <TableCell>
                       {addon.price.toLocaleString('ko-KR')}원
                     </TableCell>
                     <TableCell>
                       {PRICE_TYPE_LABEL[addon.priceType] ?? addon.priceType}
                     </TableCell>
                     <TableCell>{addon.maxDailyQty}개</TableCell>
                     <TableCell>
                       <Switch
                         checked={addon.isActive}
                         onCheckedChange={(v) => handleToggle(addon.id, v)}
                         aria-label={`${addon.name} 활성화 토글`}
                       />
                     </TableCell>
                     <TableCell className="text-right">
                       <div className="flex justify-end gap-2">
                         <Button asChild size="sm" variant="outline">
                           <Link href={`/admin/addons/${addon.id}/edit`}>
                             <Pencil className="h-3 w-3" />
                           </Link>
                         </Button>
                         <Button
                           size="sm"
                           variant="destructive"
                           onClick={() => handleDelete(addon.id)}
                         >
                           <Trash2 className="h-3 w-3" />
                         </Button>
                       </div>
                     </TableCell>
                   </TableRow>
                 ))
               )}
             </TableBody>
           </Table>
         </div>
       </div>
     )
   }
   ```
